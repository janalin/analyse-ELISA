5 Plot standard

Now letâ€™s plot the standard:

plot.std <- function() {
  ggplot(data = standard) +
    geom_point(
      aes(dilution, absorbance, color = replicate), alpha = 0.6) +
    labs(
      x = "HI titer (w/o dilution factor 1:5000)",
      y = "Absorbance (a.u.)",
      title = "IgG in standard serum"
    ) +
    scale_color_brewer(
      palette = "Set1",
      labels = paste("Replicate", seq(n.repl.std)),
      guide = guide_legend(title = "")
    ) +
    scale_y_continuous(
      limits = c(min.absorb, max.absorb + 0.1 * max.absorb)
    ) +
    scale_x_continuous(breaks = dilutions) +
    geom_hline(
      yintercept = mean(blanks$blank),
      lty = 2,
      color = "grey60",
      lwd = 0.2
    ) +
    annotate("text",
      label = "Blank",
      x = max(standard$dilution) - 0.05 * log2(standard$dilution),
      y = max(blanks$blank) + 0.2 * max(blanks$blank),
      size = 3.5,
      color = "grey50"
    ) +
    theme_plot(legend_pos = "right")
}
plot.std()

We see that the concentration of the dilution series is on a logarithmic scale. Letâ€™s replot the results with log2-transformed concentrations (log2 for a dilution factor of 2):

plot.std.log <- function() {
  ggplot(data = standard) +
    geom_point(aes(log2(dilution), absorbance, color = replicate), alpha = 0.6) +
    labs(
      x = bquote(log[2] * " HI titer (w/o dilution factor 1:5000)"),
      y = "Absorbance (a.u.)",
      title = "IgG in standard serum"
    ) +
    scale_color_brewer(
      palette = "Set1",
      labels = paste("Replicate", seq(n.repl.std)),
      guide = guide_legend(title = "")
    ) +
    scale_x_continuous(breaks = seq(4, 12, 1)) +
    geom_hline(
      yintercept = mean(blanks$blank),
      lty = 2,
      color = "grey60",
      lwd = 0.2
    ) +
    annotate("text",
      label = "Blank",
      x = max(log2(standard$dilution)) - 0.05 * max(log2(standard$dilution)),
      y = max(blanks$blank) + 10 * max(blanks$blank),
      size = 3.5,
      color = "grey50"
    ) +
    theme_plot(legend_pos = "right")
}
